#include <iostream>
using namespace std;


int main() {
	//Write a C++ program to find the maximum element in an array.

// Iteration steps for array = {1, 3, 5, 2, 6, 4}
// Step 0: maxVal = 1
// Step 1: compare 3 → maxVal = 3
// Step 2: compare 5 → maxVal = 5
// Step 3: compare 2 → maxVal = 5
// Step 4: compare 6 → maxVal = 6
// Step 5: compare 4 → maxVal = 6


	int a[] = { 1, 2, 3, 4, 5, 6 };
	int length = sizeof(a) / sizeof(a[0]); // array elements = total bytes 6*4=24 ÷ 1 element size 4 byte =24/4=6
	int maxvalue = a[0];

	int i;
	for (i = 0; i < length; i++)
	{
		if (a[i] > maxvalue)
		{
			maxvalue = a[i];
		}
		
	}
	cout <<"maxvalue is "<< maxvalue << endl;

}
