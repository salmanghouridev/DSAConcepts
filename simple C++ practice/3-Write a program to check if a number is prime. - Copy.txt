#include <iostream>
using namespace std;


int main() {
	//Write a program to check if a number is prime.

    int n;
    cin >> n;
    bool prime = (n > 1);  // assume prime if > 1

    for (int i = 2; i * i <= n && prime; i++) {
        if (n % i == 0) prime = false;
    }

    cout << (prime ? "Prime" : "Not Prime") << endl;
}
