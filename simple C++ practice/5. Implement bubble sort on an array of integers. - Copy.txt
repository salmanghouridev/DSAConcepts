#include <iostream>
using namespace std;


void bubblesort(int array[], int n) {
	for (int i = 0;i < n-1;i++) {
		for (int j=0;j < n - i - 1;j++) {
			if (array[j] > array[j + 1]) {
				int temp = array[j];
				array[j] = array[j + 1];
				array[j + 1] = temp;
			}
		}
	}
}

int main() {
	//Implement bubble sort on an array of integers.

	//Let's sort the list [4, 1, 3, 2] in ascending order: 
		//Compare 4 and 1: Since 4 > 1, swap them.The list becomes[1, 4, 3, 2].
		//Compare 4 and 3: Since 4 > 3, swap them.The list becomes[1, 3, 4, 2].
		//Compare 4 and 2: Since 4 > 2, swap them.The list becomes[1, 3, 2, 4].
	int n;
	
	int array[] = { 12, 33, 41, 23, 51, 4, 3, 2, 5, 1 };
	n = sizeof(array) / sizeof(array[0]);
	cout << "old {";
	for (int i = 0;i < n;i++) {
		
		cout<< array[i] << ",";
	}
	cout << "}";
	cout << endl;
	bubblesort(array, n);
	cout << "sorted {";
	for (int i = 0;i < n;i++) {
	
		cout << array[i] << ",";
	}
	cout << "}";
	

	
	
	

}
